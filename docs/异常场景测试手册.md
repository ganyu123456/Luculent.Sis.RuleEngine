# SIS 规则引擎异常场景测试手册

> 版本: 1.0 | 最后更新: 2026-07-18 | 适用环境: Docker Compose 部署

## 测试目标

验证规则引擎在生产环境中的异常恢复能力，确保：
1. 任何单一组件重启后系统能自动恢复
2. Worker 扩缩容时监视项正确重分区
3. 数据持久化层故障时无数据丢失
4. 配置变更不影响正在运行的计算

### 测试工具函数 (source 后可用)

```bash
# 保存为 test_helpers.sh
#!/bin/bash
RULE_ENGINE="http://localhost:11082"
MC="http://127.0.0.1:11080"

check_health() {
    echo "=== $(date '+%H:%M:%S') 健康检查 ==="
    curl -s $RULE_ENGINE/api/ruleengine/health | python3 -c "
import json,sys
d=json.load(sys.stdin)
print(f'  监视项: {d[\"monitorCount\"]}')
print(f'  活跃Worker: {d[\"activeWorkers\"]}')
for w,c in d.get('workerDistribution',{}).items():
    print(f'    {w}: {c} monitors')
"
}

check_realmonitor() {
    echo "=== $(date '+%H:%M:%S') 实时报警 ==="
    curl -s $RULE_ENGINE/api/ruleengine/alarms/realtime | python3 -c "
import json,sys
d=json.load(sys.stdin)
print(f'  活跃报警数: {len(d[\"items\"])}')
"
}

check_events_writing() {
    local monitor_key="${1:-TEST-0086}"
    echo "=== $(date '+%H:%M:%S') $monitor_key 最近事件 ==="
    curl -s -X POST "$RULE_ENGINE/api/ruleengine/alarms/history" \
        -H "Content-Type: application/json" \
        -d "{\"MonitorKeys\":[\"$monitor_key\"],\"ContainNull\":true,\"SkipCount\":0,\"MaxResultCount\":3}" \
        | python3 -c "
import json,sys
d=json.load(sys.stdin)
for i in d['items']:
    print(f'  {i[\"occurTime\"]} statusKey={i[\"statusKey\"]} lastEventName={i.get(\"lastEventName\",\"N/A\")}')
"
}

wait_for_workers() {
    local expected="${1:-2}"
    local timeout="${2:-60}"
    local elapsed=0
    while [ $elapsed -lt $timeout ]; do
        local count=$(curl -s $RULE_ENGINE/api/ruleengine/health | python3 -c "import json,sys;print(json.load(sys.stdin)['activeWorkers'])")
        if [ "$count" -ge "$expected" ]; then
            echo "  $count workers ready (${elapsed}s)"
            return 0
        fi
        sleep 2
        elapsed=$((elapsed + 2))
    done
    echo "  TIMEOUT: expected $expected workers after ${timeout}s"
    return 1
}

wait_for_monitors() {
    local expected="${1:-100}"
    local timeout="${2:-60}"
    local elapsed=0
    while [ $elapsed -lt $timeout ]; do
        local count=$(curl -s $RULE_ENGINE/api/ruleengine/health | python3 -c "import json,sys;print(json.load(sys.stdin)['monitorCount'])")
        if [ "$count" -ge "$expected" ]; then
            echo "  $count monitors loaded (${elapsed}s)"
            return 0
        fi
        sleep 2
        elapsed=$((elapsed + 2))
    done
    echo "  TIMEOUT: expected $expected monitors after ${timeout}s"
    return 1
}

check_evnetdata() {
    local monitor_key="${1:-TEST-0086}"
    echo "=== $(date '+%H:%M:%S') EvnetData $monitor_key ==="
    curl -s -X POST "$MC/api/services/monitorcenter/EvnetData/GetAllForList" \
        -H "Content-Type: application/json" \
        -H "Authorization: Bearer $JWT" \
        -d "{\"Filter\":\"$monitor_key\",\"MonitorItemIds\":[],\"ContainNull\":true,\"SkipCount\":0,\"MaxResultCount\":5}" \
        | python3 -c "
import json,sys
r=json.load(sys.stdin)['result']
print(f'  totalCount={r[\"totalCount\"]}')
for i in r['items'][:3]:
    print(f'  statusKey={i.get(\"occurMonitorStatusKey\",\"?\")} lastEventId={i.get(\"lastEventId\")} lastEventName={i.get(\"lastEventName\")}')
"
}

check_history() {
    local monitor_key="${1:-TEST-0086}"
    echo "=== $(date '+%H:%M:%S') HistoryMonitor $monitor_key ==="
    for mode in true false; do
        curl -s -X POST "$MC/api/services/monitorcenter/HistoryMonitor/GetAllForList" \
            -H "Content-Type: application/json" \
            -H "Authorization: Bearer $JWT" \
            -d "{\"Filter\":\"$monitor_key\",\"MonitorItemIds\":[],\"ContainNull\":$mode,\"SkipCount\":0,\"MaxResultCount\":3}" \
            | python3 -c "
import json,sys
r=json.load(sys.stdin)['result']
print(f'  containNull=$mode totalCount={r[\"totalCount\"]}')
for i in r['items'][:3]:
    st=i.get('startTime'); et=i.get('endTime')
    dur = f'{(et-st)/1000:.1f}s' if et else 'ongoing'
    print(f'    statusKey={i.get(\"monitorStatusKey\",\"?\")} duration={dur}')
"
    done
}
```

## 场景 7: 数据一致性验证

### 7.1 containNull 完整性

```bash
source test_helpers.sh
MONITOR="TEST-0086"

echo "=== EvnetData containNull 测试 ==="
for cn in true false; do
    echo "--- containNull=$cn ---"
    curl -s -X POST "$MC/api/services/monitorcenter/EvnetData/GetAllForList" \
        -H "Content-Type: application/json" \
        -H "Authorization: Bearer $JWT" \
        -d "{\"Filter\":\"$MONITOR\",\"MonitorItemIds\":[],\"ContainNull\":$cn,\"SkipCount\":0,\"MaxResultCount\":100}" \
        | python3 -c "
import json,sys
r=json.load(sys.stdin)['result']
hasEmpty=sum(1 for i in r['items'] if i.get('occurMonitorStatusKey','')=='')
nonEmpty=sum(1 for i in r['items'] if i.get('occurMonitorStatusKey','')!='')
print(f'  totalCount={r[\"totalCount\"]} emptyEvents={hasEmpty} nonEmptyEvents={nonEmpty}')
for i in r['items'][:2]:
    print(f'  sample: statusKey={i.get(\"occurMonitorStatusKey\",\"?\")} lastEventId={i.get(\"lastEventId\")} lastEventName={i.get(\"lastEventName\")}')
"
done

echo ""
echo "=== HistoryMonitor duration 验证 ==="
for cn in true false; do
    echo "--- containNull=$cn ---"
    curl -s -X POST "$MC/api/services/monitorcenter/HistoryMonitor/GetAllForList" \
        -H "Content-Type: application/json" \
        -H "Authorization: Bearer $JWT" \
        -d "{\"Filter\":\"$MONITOR\",\"MonitorItemIds\":[],\"ContainNull\":$cn,\"SkipCount\":0,\"MaxResultCount\":100}" \
        | python3 -c "
import json,sys
r=json.load(sys.stdin)['result']
neg=0
for i in r['items']:
    if i.get('endTime') and i['endTime'] < i['startTime']:
        neg+=1
print(f'  totalCount={r[\"totalCount\"]} negativeDurations={neg}')
if r['items']:
    i=r['items'][0]
    st=i['startTime']; et=i.get('endTime')
    dur = f'{(et-st)/1000:.1f}s' if et else 'ongoing'
    print(f'  latest: statusKey={i[\"monitorStatusKey\"]} duration={dur} (endTime={et})')
"
done
```

**预期结果**:
| 检查项 | containNull | 期望 |
|--------|-------------|------|
| emptyEvents | true | > 0 (有空状态事件) |
| emptyEvents | false | = 0 (全部过滤) |
| negativeDurations | true | 0 |
| negativeDurations | false | 0 |
| lastEventName | 两者 | 非空 (新事件, 第2个起) |

fail() { echo "  [FAIL] $1"; FAILED=$((FAILED+1)); }

echo "=========================================="
echo " SIS RuleEngine 完整回归测试"
echo " $(date)"
echo "=========================================="

# -------- 1. 健康基线 --------
echo ""
echo "--- 1. 健康基线 ---"

# -------- 2. Worker 重启恢复 --------

docker restart $TARGET
sleep 5
if wait_for_workers 2 30; then pass "Worker 重启恢复"; else fail "Worker 未恢复"; fi

# 等待事件刷新
sleep 30
